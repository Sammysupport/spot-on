131  static std::string to_hex(const std::vector<uint8_t> &vector)
132  {
133  std::stringstream stream;
134 
135  stream << std::hex;
136 
137  for(size_t i = 0, size = vector.size(); i < size; i += 1)
138  stream << std::setw(2)
139  << std::setfill('0')
140  << static_cast<int> (vector[i]);
141 
142  return stream.str();
143  }

145  static std::vector<uint8_t> from_hex(const std::string &string)
146  {
147  std::vector<uint8_t> vector;
148 
149  for(size_t i = 0, length = string.length(); i < length; i += 2)
150  {
151  auto byte = static_cast<uint8_t>
152  (strtol(string.substr(i, 2).c_str(), nullptr, 16));
153 
154  vector.push_back(byte);
155  }
156 
157  return vector;
158  }

282  uint8_t xtime(uint8_t x)
283  {
284  return static_cast<uint8_t> ((x << 1) ^ (((x >> 7) & 1) * 0x1b));
285  }
286 
287  uint8_t xtime_special(uint8_t x, uint8_t y)
288  {
289  auto xtime_y = xtime(y);
290 
291  return static_cast<uint8_t>
292  (((x & 1) * y) ^
293  (((x >> 1) & 1) * xtime_y) ^
294  (((x >> 2) & 1) * xtime(xtime_y)) ^
295  (((x >> 3) & 1) * xtime(xtime(xtime_y))) ^
296  (((x >> 4) & 1) * xtime(xtime(xtime(xtime_y)))));
297  }

427  void key_expansion(void)
428  {
429  size_t i = 0;
430 
431  while(i < m_Nk)
432  {
433  auto product = 4 * i;
434 
435  if(m_key.size() > product)
436  m_round_key[i][0] = m_key[product + 0];
437 
438  if(m_key.size() > product + 1)
439  m_round_key[i][1] = m_key[product + 1];
440 
441  if(m_key.size() > product + 2)
442  m_round_key[i][2] = m_key[product + 2];
443 
444  if(m_key.size() > product + 3)
445  m_round_key[i][3] = m_key[product + 3];
446 
447  i += 1;
448  }
449 
450  auto iterations = m_Nb * (m_Nr + 1);
451  uint8_t temp[4];
452 
453  while(i < iterations)
454  {
455  auto difference = i - 1;
456 
457  temp[0] = m_round_key[difference][0];
458  temp[1] = m_round_key[difference][1];
459  temp[2] = m_round_key[difference][2];
460  temp[3] = m_round_key[difference][3];
461 
462  if(m_Nk > 0 && i % m_Nk == 0)
463  {
464  auto quotient = i / m_Nk;
465  uint8_t t = temp[0];
466 
467  temp[0] = temp[1];
468  temp[1] = temp[2];
469  temp[2] = temp[3];
470  temp[3] = t;
471  temp[0] = s_sbox[static_cast<size_t> (temp[0])] ^
472  s_rcon[quotient][0];
473  temp[1] = s_sbox[static_cast<size_t> (temp[1])] ^
474  s_rcon[quotient][1];
475  temp[2] = s_sbox[static_cast<size_t> (temp[2])] ^
476  s_rcon[quotient][2];
477  temp[3] = s_sbox[static_cast<size_t> (temp[3])] ^
478  s_rcon[quotient][3];
479  t = 0;
480  }
481  else if(m_Nk > 0 && i % m_Nk == 4)
482  {
483  temp[0] = s_sbox[static_cast<size_t> (temp[0])];
484  temp[1] = s_sbox[static_cast<size_t> (temp[1])];
485  temp[2] = s_sbox[static_cast<size_t> (temp[2])];
486  temp[3] = s_sbox[static_cast<size_t> (temp[3])];
487  }
488 
489  difference = i - m_Nk;
490  m_round_key[i][0] = m_round_key[difference][0] ^ temp[0];
491  m_round_key[i][1] = m_round_key[difference][1] ^ temp[1];
492  m_round_key[i][2] = m_round_key[difference][2] ^ temp[2];
493  m_round_key[i][3] = m_round_key[difference][3] ^ temp[3];
494  memset(temp, 0, 4 * sizeof(temp[0]));
495  i += 1;
496  }
497  }
